func validar_numero() -> Int{
    var entrada: String? = nil
    var numeroUsuario: Int? = nil
    while true{
        print("Por favor introduce un numero:")
        entrada = readLine()
        if entrada == nil {
            print("Qué paso mi hacker, creias que habia una vulnerabilidad?")
        }
        numeroUsuario = Int(entrada!)
        if numeroUsuario == nil {
            print("Lo que introdujiste no fue un numero entero")
        }
        else{
            return numeroUsuario!
        }
    }
}
