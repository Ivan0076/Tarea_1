/*Este código usa a los maestros elementales del universo de Avatar: The Last Airbender,
con clases simples que muestran de qué nación son y qué elemento controlan.*/

// CLASE 1: Maestro Agua
class MaestroAgua {
    var nombre: String
    var nacion: String
    
    init(nombre: String, nacion: String) {
        self.nombre = nombre
        self.nacion = nacion
    }
    
    func deQueNacionEs() -> String {
        return "\(nombre) pertenece a la \(nacion)"
    }
    
    func queElementoUsa() -> String {
        return "\(nombre) controla el elemento Agua"
    }
}

// CLASE 5: Avatar 
class Avatar {
    var nombre: String
    var nacion: String
    
    init(nombre: String, nacion: String) {
        self.nombre = nombre
        self.nacion = nacion
    }
    
    func deQueNacionEs() -> String {
        return "\(nombre) nació como \(nacion), pero el Avatar pertenece a todas las naciones"
    }
    
    func queElementoUsa() -> String {
        return "\(nombre) puede usar todos los elementos: Aire, Agua, Tierra y Fuego"
    }
}
